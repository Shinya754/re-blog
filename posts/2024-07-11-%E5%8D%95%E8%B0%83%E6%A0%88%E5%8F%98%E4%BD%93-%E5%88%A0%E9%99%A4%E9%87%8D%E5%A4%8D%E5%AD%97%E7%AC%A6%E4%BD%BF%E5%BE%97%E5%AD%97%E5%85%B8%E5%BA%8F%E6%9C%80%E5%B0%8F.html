class Solution {
    public String removeDuplicateLetters(String s) {
        int n = s.length();
        Deque<Integer> stack = new LinkedList<>();
        Set<Character> set = new HashSet<>();
        Map<Character, Integer> map = new HashMap<>();
        for(int i=0;i<n;i++){
            if(!map.containsKey(s.charAt(i))){
                map.put(s.charAt(i),1);
            }
            else{
                map.put(s.charAt(i),map.get(s.charAt(i))+1);
            }
        }
        for(int i=0;i<n;i++){
            //å¦‚æœæ ˆå†…å­˜åœ¨å½“å‰å­—æ¯ï¼Œåˆ™ç›´æ¥å°†å…¶è·³è¿‡
            if(set.contains(s.charAt(i))){
                if(map.get(s.charAt(i))>1){
                    map.put(s.charAt(i), map.get(s.charAt(i))-1);
                }
                else{
                    map.remove(s.charAt(i));
                }
                continue;
            }
            //å‡ºæ ˆ
            while(stack.size() != 0 && s.charAt(stack.peek())>=s.charAt(i) && map.containsKey(s.charAt(stack.peek()))){
                int index = stack.pollFirst();
                
                set.remove(s.charAt(index));
            }
            //å…¥æ ˆ
            if(stack.size() == 0 || !set.contains(s.charAt(i))){
                stack.offerFirst(i);
                set.add(s.charAt(i));
            }
            //å°†â€œå‰©ä½™å­—æ¯â€å»æ‰å½“å‰å­—æ¯
            if(map.get(s.charAt(i))>1){
                map.put(s.charAt(i), map.get(s.charAt(i))-1);
            }
            else{
                map.remove(s.charAt(i));
            }
        }
        StringBuilder sb = new StringBuilder();
        while(stack.size()>0){
            sb.append(s.charAt(stack.pollLast()));
        }
        return sb.toString();
    }
