并查集 ​

并查集的三个主要操作：初始化、查找、合并

java
class Solution {
    int fa[];
    public int Main(int[][] graph) {
        //先建立并查集，使用一个fa[]数组记录每个节点的祖先，每一个集内，索引最小元素作为祖先
        int n = graph.length;
        fa = new int[n];
        for(int i=0;i<n;i++){
            fa[i] = i;//初始将每个节点的祖先设置为自己
        }
        //此处图路径由graph数组指示
        for(int i=0; i<n; i++){
            for(int j=i+1; j<n; j++){
                if(graph[i][j] == 1){
                    union(i, j);//在处理无向图时，合并i和j，只需要找上半三角即可，由于j一定比i大，所以小数能够作为祖先
                }
            }
        }
        for(int i=0;i<n;i++){
            find(i);
            // 最后整理一次每个并查集的路径
        }
        //到这里完成了并查集的建立
    }
    private int find(int i){
        //找到第i个元素的祖先，并优化路径，使得每个元素的父亲都作为它的祖先
        if(fa[i] == i){
            return i;
        }
        else{
            fa[i] = find(fa[i]);
            return fa[i];
        }
    }
    private void union(int left, int right){
        //将右边加入左边，左边的祖先作为新集的祖先
        int fa_left = find(left);
        int fa_right = find(right);
        fa[fa_right] = fa_left;
    }
}

如果需要更换集的祖先节点，只需要找到原祖先、新祖先，将两者的fa[]值交换，然后再对所有节点进行一次find即可。

力扣924. 尽量减少恶意软件的传播 ​

2024-04-16每日一题

作为并查集练习题是道好题，但是题干描述简直依托。

实现思路是首先用并查集整理节点，然后将每个集单独放一个Set内，并且将每个祖先额外放在一个Map中，key为这个祖先的index，value为其所在的Set的index（Set全部放一个List里面）。注意每个祖先同时存在于一个Set和Map中。

最后遍历全部的Set，找到仅一个病毒节点的Set，记录最大的那个，其中的病毒节点就是答案。

代码实现：

java
class Solution {
    int fa[];
    public int minMalwareSpread(int[][] graph, int[] initial) {
        //先建立并查集，使用一个fa[]数组记录每个节点的祖先，每一个集内，索引最小元素作为祖先
        int n = graph.length;
        fa = new int[n];
        for(int i=0;i<n;i++){
            fa[i] = i;//初始将每个节点的祖先设置为自己
        }
        for(int i=0; i<n; i++){
            for(int j=i+1; j<n; j++){
                if(graph[i][j] == 1){
                    union(i, j);//合并i和j，只需要找上半三角即可，由于j一定比i大，所以小数能够作为祖先
                }
            }
        }
        for(int i=0;i<n;i++){
            find(i);
            // System.out.print(fa[i]);
            // System.out.print(" ");
        }
        // System.out.println();
        //到这里完成了并查集的建立
        int created_Sets = 0;
        List<Set<Integer>> list = new ArrayList<>();
        Map<Integer, Integer> accessed_nodes = new HashMap<>();
        for(int i=0;i<n;i++){
            if(!accessed_nodes.containsKey(fa[i])){
                //i的祖先没被访问过，说明i就是一个祖先，为其创建一个set并且加入已访问map，key为其自己，value为那个set的索引
                Set<Integer> tempSet = new HashSet<>();
                tempSet.add(fa[i]);
                tempSet.add(i);
                list.add(tempSet);

                accessed_nodes.put(fa[i], created_Sets);
                created_Sets++;
            }
            else{
                //i的祖先被访问过，则加入其祖先所在的那个set
                int setIndex = accessed_nodes.get(fa[i]);
                Set<Integer> tempSet = list.get(setIndex);
                tempSet.add(i);
                // System.out.println(i+" "+setIndex);
            }
        }
        //到这里将所有节点放入set和map中
        Set<Integer> virusSet = new HashSet<>();
        for(int num:initial){
            virusSet.add(num);
            // System.out.println(num);
        }
        //对每一个set，统计病毒节点个数，当其恰好为1时记录这个set的大小
        int ans = 114514;
        int biggestSize = 0;
        for(Set<Integer> set : list){
            
            System.out.println();
            int virusCount=0;
            int virusNode = -1;
            for(int num: set){
                System.out.println(num);
                if(virusSet.contains(num)){
                    
                    System.out.println("****");
                    virusCount++;
                    if(virusCount>1){
                        break;
                    }
                    virusNode = num;
                }
            }
            if(virusCount == 1){
                if(set.size()>biggestSize){
                    biggestSize = set.size();
                    ans = virusNode;
                }
                if(set.size()==biggestSize){
                    ans = Math.min(ans, virusNode);
                }
            }
        }
        Arrays.sort(initial);
        if(ans == 114514) ans = initial[0];
        return ans;
    }
    private int find(int i){
        //找到第i个元素的祖先，并优化路径，使得每个元素的父亲都作为它的祖先
        if(fa[i] == i){
            return i;
        }
        else{
            fa[i] = find(fa[i]);
            return fa[i];
        }
    }
    private void union(int left, int right){
        //将右边加入左边，左边的祖先作为新集的祖先
        int fa_left = find(left);
        int fa_right = find(right);
        fa[fa_right] = fa_left;
    }
}

复杂度分析：

时间复杂度O(n2)O(n^2)O(n2)，建立并查集最耗时间。

空间复杂度应该是O(n)O(n)O(n)，如果java中哈希表会预先开辟一大块空间则比O(n)O(n)O(n)大，不过只看已使用空间的话就是O(n)O(n)O(n)
